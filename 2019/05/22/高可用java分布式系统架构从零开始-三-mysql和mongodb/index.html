一 mysql

如果使用了阿里云的apt镜像源, 直接通过apt安装

tar -xvf (文件名).tar.gz

groupadd mysql
useradd -r -g mysql mysql

mkdir data

chown -R root .
chown -R mysql:mysql data

./mysqld –basedir=/home/linco/app/mysql –datadir=/home/linco/app/mysql/data –initialize

sudo vim /etc/my.cnf

[mysqld]
basedir=/home/linco/app/mysql
datadir=/home/linco/app/mysql/data
socket=/home/linco/app/mysql/tmp/mysql.sock
port=3306
sql_mode=NO_ENGINE_SUBSTITUTION,STRICT_TRANS_TABLES
user=mysql

[client]
socket=/home/linco/app/mysql/tmp/mysql.sock

[mysqld_safe]

user=mysql

cd ~/app/mysql/data
mkdir tmp
chown -R mysql:mysql tmp

cp ~/app/mysql/support-files/mysql.server /etc/init.d/mysql
sudo systemctl enable mysql.service

service mysql start

mysql -u root -p

use mysql;

update user set host=’%’ where user = “root”;

alter user ‘root‘@’%’ identified by’your password’;

flush privileges;

ps -ef|grep mysql
service mysql status
mysqld –user=mysql
